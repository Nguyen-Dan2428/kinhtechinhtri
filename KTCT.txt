<script>

function showSection(id){
    document.getElementById('sim').style.display='none';
    document.getElementById('game').style.display='none';
    document.getElementById(id).style.display='block';
}

let questions=[
{
q:"Quyền sở hữu chưa rõ ràng?",
options:["Tăng pháp luật","Giảm nhà nước","Bỏ thị trường"],
correct:0,
explain:"Cần pháp luật rõ ràng"
},
{
q:"Doanh nghiệp yếu?",
options:["Độc quyền","Cạnh tranh","Đóng cửa"],
correct:1,
explain:"Cạnh tranh giúp phát triển"
}
];

let current=0,score=0,userAnswers=[];

function loadQuestion(){
let q=questions[current];

document.getElementById("question").innerHTML=`
<b>${q.q}</b>

<div class="option-row">
<button class="option-btn" onclick="answer(0)">A. ${q.options[0]}</button>
<button class="option-btn" onclick="answer(1)">B. ${q.options[1]}</button>
<button class="option-btn" onclick="answer(2)">C. ${q.options[2]}</button>
</div>
`;
}

function answer(i){
userAnswers.push(i);
if(i===questions[current].correct) score++;
current++;
if(current<questions.length) loadQuestion();
else showResult();
}

function showResult(){
document.getElementById("question").innerHTML="";
document.getElementById("gameResult").innerHTML=
`<h3>Kết quả: ${score}/${questions.length}</h3>`;
}

loadQuestion();
